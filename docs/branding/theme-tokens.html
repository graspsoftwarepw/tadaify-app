CSS Custom Properties

/* Light mode (default) */
:root, [data-theme="light"] {
  /* Surface */
  --background:          #F9FAFB;
  --surface:             #FFFFFF;
  --surface-elevated:   #FFFFFF;  /* differentiate via shadow-md */
  --border-subtle:       #E5E7EB;
  --border-strong:       #D1D5DB;

  /* Foreground */
  --text-primary:        #111827;
  --text-secondary:      #374151;
  --text-muted:          #6B7280;
  --text-inverse:        #F9FAFB;

  /* Brand */
  --brand-primary:         #6366F1;
  --brand-primary-hover:   #4F46E5;
  --brand-primary-pressed: #4338CA;
  --brand-secondary:       #8B5CF6;
  --brand-warm:            #F59E0B;

  /* Semantic */
  --success:  #10B981;  --success-bg: #D1FAE5;  --success-text: #065F46;
  --warning:  #F59E0B;  --warning-bg: #FEF3C7;  --warning-text: #92400E;
  --danger:   #EF4444;  --danger-bg:  #FEE2E2;  --danger-text:  #991B1B;
  --info:     #3B82F6;  --info-bg:    #DBEAFE;  --info-text:    #1E40AF;

  /* Gradients */
  --hero-gradient:      linear-gradient(135deg, #6366F1, #8B5CF6);
  --warm-glow-gradient: linear-gradient(135deg, #F59E0B, #FDE68A);
}

/* Dark mode */
[data-theme="dark"] {
  /* Surface — cool desaturated navy */
  --background:          #0B0F1E;
  --surface:             #141828;
  --surface-elevated:   #1E2338;
  --border-subtle:       #252A3D;
  --border-strong:       #323754;

  --text-primary:        #F1F5F9;
  --text-secondary:      #CBD5E1;
  --text-muted:          #64748B;
  --text-inverse:        #111827;

  /* Brand — lifted for WCAG AA on dark canvas */
  --brand-primary:         #818CF8;  /* 4.7:1 on #0B0F1E */
  --brand-primary-hover:   #A5B4FC;
  --brand-primary-pressed: #6366F1;
  --brand-secondary:       #A78BFA;
  --brand-warm:            #FCD34D;  /* 9.1:1 on #0B0F1E */

  --success:  #34D399;  --success-bg: #064E3B;  --success-text: #6EE7B7;
  --warning:  #FCD34D;  --warning-bg: #451A03;  --warning-text: #FDE68A;
  --danger:   #F87171;  --danger-bg:  #450A0A;  --danger-text:  #FCA5A5;
  --info:     #60A5FA;  --info-bg:    #1E3A5F;  --info-text:    #BFDBFE;

  --hero-gradient:      linear-gradient(135deg, #6366F1, #7C3AED);
  --warm-glow-gradient: linear-gradient(135deg, #F59E0B, #FDE68A);
}
